• Intro to ES6

Default Parameters

es5

                
                  function(x,y) {
                    var a = x || 1;
                    var b = y || 1;
                    return a+b;
                  }

Output:

es6

                
                  function(x=1,y=1) {
                    return x+y;
                  }

String Literals

es5

              
                var first = "Rupesh";
                var last = "B";
                var text = "My Name is" + " "+ first + " " + last;

Output:

es6

              
                var first = "Rupesh";
                var last = "B";
                var text = `My Name is ${first} ${last}`;

Class System

es5

              
                

                  let PersonES5 = function(name) {
                      this.name = name;
                      this.getName = function() {
                          return this.name;
                      };
                  };

Output:

es6

              
                

                  class PersonES6 {
                      constructor(name) {
                          this.name = name;
                      }

                      getName() {
                          return this.name;
                      }
                  }

Scoping

es5

              
                const es5Scoping = function () {
                    if(true) {
                        var name = "Rupesh";
                    }
                    console.log("es5-scoping",name);
                    return name;
                };

Output:

es6

              
                const es6Scoping = function() {
                    if(true) {
                        let name = "Rupesh";
                    }
                    console.log("es6-scoping",name);
                    return name;
                };

Destructuring

es5

              
                var person = {name: "Rupesh",age: 25,rank: 1};
                var name = person.name;
                var age = person.age;

Output:

es6

              
                let person = {name: "Rupesh",age: 25,rank: 1};
                let {name, age} = person;

Spread Operator

es5

              
                var arr1 = [1,2], arr2 = [3,4];
                arr1.concat(arr2);
                //arr1 is [1,2,3,4]

Output:

es6

              
                let  arr2 = [3,4];
                let arr1 = [1, 2, ...arr2];
                //arr1 is [1,2,3,4]
